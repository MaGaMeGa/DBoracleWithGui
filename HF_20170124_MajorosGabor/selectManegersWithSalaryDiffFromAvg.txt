SELECT First_name||' '||last_name as empName, salary FROM EMPLOYEES
WHERE salary >(SELECT AVG(SALARY)-AVG(SALARY)*0.1 FROM (
SELECT First_name||' '||last_name as empName, EMPLOYEE_ID, salary FROM EMPLOYEES
WHERE EMPLOYEE_ID IN (SELECT MANAGER_ID FROM EMPLOYEES)))
AND salary <(SELECT AVG(SALARY)+AVG(SALARY)*0.1 FROM (
SELECT First_name||' '||last_name as empName, EMPLOYEE_ID, salary FROM EMPLOYEES
WHERE EMPLOYEE_ID IN (SELECT MANAGER_ID FROM EMPLOYEES)))
ORDER BY SALARY
